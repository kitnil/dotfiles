* Installation

export KV_RELEASE=v0.53.0
kubectl apply -f https://github.com/kubevirt/kubevirt/releases/download/${KV_RELEASE}/kubevirt-operator.yaml
kubectl apply -f https://github.com/kubevirt/kubevirt/releases/download/${KV_RELEASE}/kubevirt-cr.yaml

#+begin_src yaml
  apiVersion: kubevirt.io/v1
  kind: VirtualMachine
  metadata:
    name: testvm
  spec:
    running: true
    template:
      metadata:
        labels:
          kubevirt.io/size: small
          kubevirt.io/domain: testvm
      spec:
        domain:
          devices:
            disks:
              - name: containerdisk
                disk:
                  bus: virtio
              - name: cloudinitdisk
                disk:
                  bus: virtio
          resources:
            requests:
              memory: 1024M
        volumes:
          - name: containerdisk
            containerDisk:
              image: quay.io/kubevirt/fedora-cloud-container-disk-demo
          - cloudInitNoCloud:
              userData: |-
                #cloud-config
                password: fedora
                chpasswd: { expire: False }
            name: cloudinitdisk
#+end_src
